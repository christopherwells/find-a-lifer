<project_specification>
  <project_name>Find-A-Lifer</project_name>

  <overview>
    Find-A-Lifer is an interactive web application that helps birders discover and plan trips to find "life birds" — species they've never seen before. By combining eBird Status &amp; Trends abundance and occurrence data with an interactive map, users can explore where target species are located by week, assess probability of finding them, compare destinations, and plan birding trips. The app is designed for both casual birders and serious listers pursuing goals like seeing every native North American bird species. Originally built as an R Shiny app, this is a full rewrite targeting a modern Python backend with a JavaScript frontend, architected for minimal server resources with heavy client-side computation, browser-based data persistence, and a data pipeline designed to run on HPC clusters.
  </overview>

  <database_schema>
    <note>No server-side database. All data is pre-processed static files served to the client. User data is stored in browser IndexedDB.</note>
    <static_data_files>
      <species_metadata>
        - species_id (integer, unique)
        - speciesCode (string, eBird code)
        - comName (string, common name)
        - sciName (string, scientific name)
        - familyComName (string, family common name)
        - taxonOrder (integer, for sorting)
        - invasionStatus (string: Native | Introduced | Rare/Accidental | Unknown)
        - conservStatus (string: Least Concern | Near Threatened | Vulnerable | Critically Endangered | Unknown)
        - difficultyScore (float, 0-1, derived from avg occurrence probability)
        - difficultyLabel (string: Easy | Moderate | Hard | Very Hard)
        - isRestrictedRange (boolean)
        - ebirdUrl (string, link to eBird species page)
        - photoUrl (string, Macaulay Library thumbnail URL if available)
      </species_metadata>
      <weekly_occurrence_files>
        - 52 files (week_01.json through week_52.json)
        - Each contains array of {cell_id, species_id, probability} records
        - probability is float 0-1 representing occurrence likelihood
      </weekly_occurrence_files>
      <grid_data>
        - Grid cell definitions with cell_id, center lat/lng, and bounding polygon
        - Format optimized for MapLibre rendering (GeoJSON or vector tiles)
        - 27km resolution (architecture supports future 3km upgrade)
      </grid_data>
      <regions_data>
        - Region polygons (GeoJSON) for map display and filtering
        - Species-region mapping for checklist filtering
      </regions_data>
    </static_data_files>
    <client_side_storage>
      <life_list>
        - IndexedDB store: "lifeList"
        - Keys: speciesCode (string)
        - Values: {speciesCode, comName, dateAdded, source: "manual" | "import"}
      </life_list>
      <user_preferences>
        - IndexedDB store: "preferences"
        - Keys: setting name
        - Values: {darkMode, defaultRegion, lastWeek, activeFilters}
      </user_preferences>
    </client_side_storage>
  </database_schema>
